BarLayout 数据格式

类型定义 

{
    // 坐标系
   coordinate: {
       x: number // 坐标系左下角横坐标
       y: number // 坐标系左下角纵坐标
       width: number // 坐标系的宽
       height: number // 坐标系的高
       xAxis: categoryAxios | valueAxios // x坐标轴
       yAxis: categoryAxios | valueAxios // y坐标轴
   }

   // 直方矩形
   node: Array<{
       name?: string // 分类名称（如果是一维的，就不存在此项）
       bar: Array<{ // 当前分类的一个个直方矩形
           value: number // 矩形原始的值
           x: number // 矩形左下角横坐标
           y: number // 矩形左下角纵坐标
           width: number // 矩形宽
           height: number // 矩形高
       }>
   }>

}

categoryAxios 

{
    type: "category"
    data: Array<string>
}

valueAxios 

{
    type: "value"
    data: Array<number>
}
